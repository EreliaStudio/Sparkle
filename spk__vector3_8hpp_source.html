439  IVector3<TType> rotate(const IVector3<TType> &p_rotationValues) const
440  {
441  // Conversion of rotation values from degrees to radians and calculation of sine and cosine for each
442  TType cos_x = std::cos(spk::degreeToRadian(p_rotationValues.x));
443  TType sin_x = std::sin(spk::degreeToRadian(p_rotationValues.x));
444  TType cos_y = std::cos(spk::degreeToRadian(p_rotationValues.y));
445  TType sin_y = std::sin(spk::degreeToRadian(p_rotationValues.y));
446  TType cos_z = std::cos(spk::degreeToRadian(p_rotationValues.z));
447  TType sin_z = std::sin(spk::degreeToRadian(p_rotationValues.z));
448 
449  // Application of rotation matrix
450  IVector3<TType> result;
451  result.x = cos_y * cos_z * x + (sin_x * sin_y * cos_z - cos_x * sin_z) * y + (cos_x * sin_y * cos_z + sin_x * sin_z) * z;
452  result.y = cos_y * sin_z * x + (sin_x * sin_y * sin_z + cos_x * cos_z) * y + (cos_x * sin_y * sin_z - sin_x * cos_z) * z;
453  result.z = -sin_y * x + sin_x * cos_y * y + cos_x * cos_y * z;
454 
455  return result;
456  }
