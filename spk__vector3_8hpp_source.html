372  IVector3<TType> rotate(const IVector3<TType> &p_rotationValues) const
373  {
374  // Conversion of rotation values from degrees to radians and calculation of sine and cosine for each
375  TType cos_x = std::cos(spk::degreeToRadian(p_rotationValues.x));
376  TType sin_x = std::sin(spk::degreeToRadian(p_rotationValues.x));
377  TType cos_y = std::cos(spk::degreeToRadian(p_rotationValues.y));
378  TType sin_y = std::sin(spk::degreeToRadian(p_rotationValues.y));
379  TType cos_z = std::cos(spk::degreeToRadian(p_rotationValues.z));
380  TType sin_z = std::sin(spk::degreeToRadian(p_rotationValues.z));
381 
382  // Application of rotation matrix
383  IVector3<TType> result;
384  result.x = cos_y * cos_z * x + (sin_x * sin_y * cos_z - cos_x * sin_z) * y + (cos_x * sin_y * cos_z + sin_x * sin_z) * z;
385  result.y = cos_y * sin_z * x + (sin_x * sin_y * sin_z + cos_x * cos_z) * y + (cos_x * sin_y * sin_z - sin_x * cos_z) * z;
386  result.z = -sin_y * x + sin_x * cos_y * y + cos_x * cos_y * z;
387 
388  return result;
389  }
