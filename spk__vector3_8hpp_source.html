368  IVector3<TType> rotate(const IVector3<TType> &p_rotationValues) const
369  {
370  // Conversion of rotation values from degrees to radians and calculation of sine and cosine for each
371  TType cos_x = std::cos(spk::degreeToRadian(p_rotationValues.x));
372  TType sin_x = std::sin(spk::degreeToRadian(p_rotationValues.x));
373  TType cos_y = std::cos(spk::degreeToRadian(p_rotationValues.y));
374  TType sin_y = std::sin(spk::degreeToRadian(p_rotationValues.y));
375  TType cos_z = std::cos(spk::degreeToRadian(p_rotationValues.z));
376  TType sin_z = std::sin(spk::degreeToRadian(p_rotationValues.z));
377 
378  // Application of rotation matrix
379  IVector3<TType> result;
380  result.x = cos_y * cos_z * x + (sin_x * sin_y * cos_z - cos_x * sin_z) * y + (cos_x * sin_y * cos_z + sin_x * sin_z) * z;
381  result.y = cos_y * sin_z * x + (sin_x * sin_y * sin_z + cos_x * cos_z) * y + (cos_x * sin_y * sin_z - sin_x * cos_z) * z;
382  result.z = -sin_y * x + sin_x * cos_y * y + cos_x * cos_y * z;
383 
384  return result;
385  }
