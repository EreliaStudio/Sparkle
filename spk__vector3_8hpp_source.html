384  IVector3<TType> rotate(const IVector3<TType> &p_rotationValues) const
385  {
386  // Conversion of rotation values from degrees to radians and calculation of sine and cosine for each
387  TType cos_x = std::cos(spk::degreeToRadian(p_rotationValues.x));
388  TType sin_x = std::sin(spk::degreeToRadian(p_rotationValues.x));
389  TType cos_y = std::cos(spk::degreeToRadian(p_rotationValues.y));
390  TType sin_y = std::sin(spk::degreeToRadian(p_rotationValues.y));
391  TType cos_z = std::cos(spk::degreeToRadian(p_rotationValues.z));
392  TType sin_z = std::sin(spk::degreeToRadian(p_rotationValues.z));
393 
394  // Application of rotation matrix
395  IVector3<TType> result;
396  result.x = cos_y * cos_z * x + (sin_x * sin_y * cos_z - cos_x * sin_z) * y + (cos_x * sin_y * cos_z + sin_x * sin_z) * z;
397  result.y = cos_y * sin_z * x + (sin_x * sin_y * sin_z + cos_x * cos_z) * y + (cos_x * sin_y * sin_z - sin_x * cos_z) * z;
398  result.z = -sin_y * x + sin_x * cos_y * y + cos_x * cos_y * z;
399 
400  return result;
401  }
