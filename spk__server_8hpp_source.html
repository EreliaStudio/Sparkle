221  bool receiveFromClient(ClientID p_clientId, SOCKET p_socket)
222  {
223  auto message = messagePool.obtain();
224 
225  const int headerSize = sizeof(spk::Message::Header);
226  int bytesRead = ::recv(p_socket, reinterpret_cast<char*>(&message->header()), headerSize, 0);
227  if (bytesRead == headerSize)
228  {
229  if (message->header().length() > 0)
230  {
231  message->resize(message->header().length());
232 
233  char* dataBuffer = reinterpret_cast<char*>(message->buffer().data());
234  size_t totalBytesReceived = 0;
235  while (totalBytesReceived < message->header().length())
236  {
237  bytesRead = ::recv(p_socket, dataBuffer + totalBytesReceived, message->header().length() - totalBytesReceived, 0);
238  if (bytesRead <= 0)
239  {
240  return false;
241  }
242  totalBytesReceived += bytesRead;
243  }
244  }
245  message->header().emitterID = p_clientId;
246  messageQueue.push_back(std::move(message));
247  return true;
248  }
249  else if (bytesRead < 0)
250  {
251  return false;
252  }
253  else if (bytesRead == 0)
254  {
255  return false;
256  }
