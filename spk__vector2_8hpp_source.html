550  template <typename TOtherType, typename = typename std::enable_if<std::is_arithmetic<TOtherType>::value>::type>
551  IVector2 rotate(const IVector2<TOtherType>& p_center, const float& p_angle) const
552  {
553  float theta = spk::degreeToRadian(p_angle);
554  float pcos = std::cos(theta);
555  float psin = std::sin(theta);
556  IVector2 result;
557  result.x = (x - p_center.x) * pcos - (y - p_center.y) * psin;
558  result.y = (x - p_center.x) * psin + (y - p_center.y) * pcos;
559  return (result + p_center);
560  }
