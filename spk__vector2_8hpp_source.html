498  template <typename TOtherType, typename = typename std::enable_if<std::is_arithmetic<TOtherType>::value>::type>
499  IVector2 rotate(const IVector2<TOtherType>& p_center, const float& p_angle) const
500  {
501  float theta = spk::degreeToRadian(p_angle);
502  float pcos = std::cos(theta);
503  float psin = std::sin(theta);
504  IVector2 result;
505  result.x = (x - p_center.x) * pcos - (y - p_center.y) * psin;
506  result.y = (x - p_center.x) * psin + (y - p_center.y) * pcos;
507  return (result + p_center);
508  }
